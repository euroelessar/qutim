CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( LIBQUTIM_DOC )

SET( LIBQUTIM_DOC_LANG English CACHE STRING "Documentation language")
SET( LIBQUTIM_DOC_BRIEF_MEMBER_DESC NO CACHE STRING "Include brief member descriptions after the members that are listed in the file and class documentation" )
SET( LIBQUTIM_DOC_EXTRACT_PRIVATE NO CACHE STRING "" )
SET( LIBQUTIM_DOC_EXTRACT_STATIC NO CACHE STRING "" )
SET( LIBQUTIM_DOC_QUIET NO CACHE STRING "Quiet documentation generation" )
SET( LIBQUTIM_DOC_FILE_PATTERNS *.h CACHE STRING "Space separated wildcard patterns" )
SET( LIBQUTIM_DOC_EXCLUDE_PATTERNS *_p.h CACHE STRING "Space separated wildcard patterns" )
SET( LIBQUTIM_DOC_EXCLUDE_SYMBOLS d_ptr CACHE STRING "Space separated wildcard patterns" )
SET( LIBQUTIM_DOC_SOURCE_BROWSER NO CACHE STRING "" )
SET( LIBQUTIM_DOC_INLINE_SOURCES NO CACHE STRING "" )
SET( LIBQUTIM_DOC_REFERENCED_BY_RELATION NO CACHE STRING "For each documented function all documented functions referencing it will be listed" )
SET( LIBQUTIM_DOC_REFERENCES_RELATION NO CACHE STRING "For each documented function all documented entities called/used by that function will be listed" )
SET( LIBQUTIM_DOC_GENERATE_ECLIPSEHELP NO CACHE STRING "" )
SET( LIBQUTIM_DOC_GENERATE_TREEVIEW YES CACHE STRING "" )
SET( LIBQUTIM_DOC_TREEVIEW_WIDTH 250 CACHE STRING "Set width if GENERATE_TREEVIEW is YES" )
SET( LIBQUTIM_DOC_EXPAND_ONLY_PREDEF NO CACHE STRING "" )
SET( LIBQUTIM_DOC_CLASS_DIAGRAMS YES CACHE STRING "" )
SET( LIBQUTIM_DOC_HIDE_UNDOC_RELATIONS YES CACHE STRING "" )
SET( LIBQUTIM_DOC_HAVE_DOT NO CACHE STRING "Advanced graph visualization" )
SET( LIBQUTIM_DOC_UML_LOOK NO CACHE STRING "" )
SET( LIBQUTIM_DOC_CALL_GRAPH NO CACHE STRING "Works only if HAVE_DOT is YES. Enable only if you know what you are doing!" )
SET( LIBQUTIM_DOC_CALLER_GRAPH NO CACHE STRING "Works only if HAVE_DOT is YES. Enable only if you know what you are doing!" )
SET( LIBQUTIM_DOC_DOT_GRAPH_MAX_NODES 50 CACHE STRING "" )
SET( LIBQUTIM_DOC_MAX_DOT_GRAPH_DEPTH 0 CACHE STRING "0 is equal to no restriction" )

IF ( LIBQUTIM_DOC_SOURCE_BROWSER STREQUAL YES )
   SET (LIBQUTIM_DOC_LAYOUT_TAB_GUARD_GL_OP "")
   SET (LIBQUTIM_DOC_LAYOUT_TAB_GUARD_GL_ED "")
ELSE ()
   SET (LIBQUTIM_DOC_LAYOUT_TAB_GUARD_GL_OP "<!--")
   SET (LIBQUTIM_DOC_LAYOUT_TAB_GUARD_GL_ED "-->")
ENDIF ()

MARK_AS_ADVANCED(
   LIBQUTIM_DOC_BRIEF_MEMBER_DESC
   LIBQUTIM_DOC_EXTRACT_PRIVATE
   LIBQUTIM_DOC_EXTRACT_STATIC
   LIBQUTIM_DOC_QUIET
   LIBQUTIM_DOC_FILE_PATTERNS
   LIBQUTIM_DOC_EXCLUDE_PATTERNS
   LIBQUTIM_DOC_EXCLUDE_SYMBOLS
   LIBQUTIM_DOC_INLINE_SOURCES
   LIBQUTIM_DOC_REFERENCED_BY_RELATION
   LIBQUTIM_DOC_REFERENCES_RELATION
   LIBQUTIM_DOC_TREEVIEW_WIDTH
   LIBQUTIM_DOC_EXPAND_ONLY_PREDEF
   LIBQUTIM_DOC_HIDE_UNDOC_RELATIONS
   LIBQUTIM_DOC_CALL_GRAPH
   LIBQUTIM_DOC_CALLER_GRAPH
   LIBQUTIM_DOC_DOT_GRAPH_MAX_NODES
   LIBQUTIM_DOC_MAX_DOT_GRAPH_DEPTH
)
